# Drawer

export const OpenButton = () => {
  const [open, setOpen] = useState(false);
  const [side, setSide] = useState('left');
  return (
    <>
        <Drawer isOpen={open} onClose={() => setOpen(false)} title="Basic Drawer" placement={side}>
            <p>Drawer content</p>
            <p>Some contents...</p>
            <p>Some contents...</p>
            <p>Some contents...</p>
        </Drawer>
        <ButtonGroup>
            <Button onClick={() => {setSide('left');setOpen(open => !open);}}>Open Left</Button>
            <Button onClick={() => {setSide('right');setOpen(open => !open);}}>Open Right</Button>
            <Button onClick={() => {setSide('top');setOpen(open => !open);}}>Open Top</Button>
            <Button onClick={() => {setSide('bottom');setOpen(open => !open);}}>Open Bottom</Button>
        </ButtonGroup>
    </>
  )
}

### Properties

|Property|Type|Default Value|Description|Required|
|---|---|---|---|---|
|children|React.ReactNode|-|Drawer content|false|
|isOpen|boolean|false|Open drawer|false|
|onClose|() => void|undefined|Close drawer|false|
|placement|'left' \| 'right' \| 'top' \| 'bottom'|'left'|Drawer placement|false|
|size|'small' \| 'medium' \| 'fullscreen' \| string \| number|'medium'|Drawer size|false|
|title|React.ReactNode|-|Drawer title|false|
|closeIcon|React.ReactNode|-|Close icon|false|
|testingID| string | - | The id used for testing purposes.<br/>`<div data-testid="my-test-id"/>` |No|
